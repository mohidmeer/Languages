# Operations On Numbers

``` javascript lineons
    let age = 30;

    // Absolute value
    let absValue = Math.abs(age);
    console.log(absValue); // Output: 30

    // Rounding to the nearest integer
    let roundedValue = Math.round(age);
    console.log(roundedValue); // Output: 30

    // Rounding down to the nearest integer
    let floorValue = Math.floor(age);
    console.log(floorValue); // Output: 30

    // Rounding up to the nearest integer
    let ceilValue = Math.ceil(age);
    console.log( ceilValue); // Output: 30

    // Truncate decimal part
    let truncValue = Math.trunc(age);
    console.log( truncValue); // Output: 30

    // Square root
    let sqrtValue = Math.sqrt(age);
    console.log(sqrtValue); // Output: 5.477225575051661

    // Cube root
    let cbrtValue = Math.cbrt(age);
    console.log("Cube root:", cbrtValue); // Output: 3.1072325059538586

    // Random number between 0 and 1
    let randomValue = Math.random();
    console.log("Random number between 0 and 1:", randomValue); // Output: (random value between 0 and 1, e.g., 0.723456789)

```
